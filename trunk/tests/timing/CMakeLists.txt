  external_vec_et1
  )

# Hand-coded C matrix tests:

  matrix_c1
  matrix_c2
  )

# Fixed-matrix expression template tests:
SET(FIXED_MAT_TESTS
  fixed_mat_et1
  fixed_mat_et2

  dynamic_mat_et1
  )

  external_mat_et1

  )

  ${DYNAMIC_VEC_TESTS}
  ${EXTERNAL_VEC_TESTS}
  ${C_MAT_TESTS}
  ${FIXED_MAT_TESTS}
  ${DYNAMIC_MAT_TESTS}
  ${EXTERNAL_MAT_TESTS}
  ${C_MATVEC_TESTS}
  ${FIXED_MATVEC_TESTS}
  ${DYNAMIC_MATVEC_TESTS}
  ${EXTERNAL_MATVEC_TESTS}
  )

FOREACH(Test ${TimingTests})
  ADD_EXECUTABLE(${Test} ${Test}.cpp)
